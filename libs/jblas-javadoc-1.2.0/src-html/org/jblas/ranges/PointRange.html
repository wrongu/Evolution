042    package org.jblas.ranges;
043    
044    /**
045     * A PointRange is a range which only has a single point.
046     */
047    public class PointRange implements Range {
048        private int value;
049        private boolean consumed; 
050    
051        /** Construct a new PointRange with the one given index. */
052        public PointRange(int v) {
053            value = v;
054        }
055        
056        public void init(int l, int u) {
057            consumed = false;
058        }
059    
060        public int length() {
061            return 1;
062        }
063        
064        public int value() {
065            return value;
066        }
067        
068        public int index() {
069            return 0;
070        }
071        
072        public void next() {
073            consumed = true;
074        }
075        
076        public boolean hasMore() {
077            return !consumed;
078        }
079    }
