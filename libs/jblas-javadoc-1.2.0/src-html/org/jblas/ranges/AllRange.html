042    package org.jblas.ranges;
043    
044    import org.jblas.*;
045    
046    /**
047     * A range over all available indices. Can be used to address whole columns or rows. Like
048     * the ":" index in matlab. Don't forget to call init() before using this range.
049     */
050    public class AllRange implements Range {
051        private int lower;
052        private int upper;
053        private int value;
054        private int counter;
055        
056        public AllRange() {}
057        
058        public void init(int l, int u) {
059            lower = l;
060            upper = u;
061            value = l;
062            counter = 0;
063        }
064        
065        public int length() {
066            return upper - lower;
067        }
068        
069        public int value() {
070            return value;
071        }
072        
073        public int index() {
074            return counter;
075        }
076        
077        public void next() {
078            counter++;
079            value++;
080        }
081        
082        public boolean hasMore() {
083            return value < upper;
084        }
085    }
