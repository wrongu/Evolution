037    package org.jblas.exceptions;
038    
039    /**
040     * LapackException for a specific argument. LAPACK routines routinely check
041     * whether some arguments contain illegal arguments. This exception class
042     * automatically constructs a message for a given argument index.
043     */
044    public class LapackArgumentException extends LapackException {
045        /** 
046         * Construct exception for given function and info. Message
047         * will read "Argument <info> had an illegal value.");
048         */
049            public LapackArgumentException(String function, int info) {
050                    super(function, "Argument " + info + " had an illegal value.");
051            }
052    
053        public LapackArgumentException(String function, String message) {
054            super(function, message);
055        }
056    }
